rule "CONTAINS_TREE_NUTS_COMBINATION"
	when
		Start_Parentheses
			has ingredient with contains any HIG CONTAINS_TREE_NUTS_ALLERGEN_HIG
			Or
			contains any of these attributes CONTAINS_TREE_NUTS_CLAIM_BASE,CONTAINS_ALMOND_CLAIM_BASE,CONTAINS_BRAZIL_NUT_CLAIM_BASE,CONTAINS_CASHEW_CLAIM_BASE,CONTAINS_HAZELNUT_CLAIM_BASE,CONTAINS_PECAN_NUT_CLAIM_BASE,CONTAINS_PISTACHIO_CLAIM_BASE,CONTAINS_SHEA_NUT_CLAIM_BASE,CONTAINS_WALNUTS_CLAIM_BASE,CONTAINS_MACADAMIA_NUTS_CLAIM_BASE,CONTAINS_COCONUT_CLAIM_BASE
		End_Parentheses
		And
		has none of these attributes TREE_NUT_FREE_CLAIM_BASE, TREE_NUT_FREE_FACILITY_CLAIM_BASE, TREE_NUT_FREE_EQUIPMENT_CLAIM_BASE
	then
		The product gets attribute CONTAINS_TREE_NUTS_COMBINATION
end
